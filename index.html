// ======================= // 🚗 Servis32 - Server.js (final) // ======================= const express = require("express"); const path = require("path"); const app = express(); // Middleware për body parsing app.use(express.json()); app.use(express.urlencoded({ extended: true })); // 📁 Shërbe skedarët statikë (index.html, dashboard.html, etj.) app.use(express.static(path.join(__dirname))); // ✅ API për login (përputhet me fetch('/api/login')) app.post("/api/login", (req, res) => { const { user, pass } = req.body; const validUser = "admin"; const validPass = "1234"; if (user === validUser && pass === validPass) { console.log("✔️ Login sukses!"); // mund të dërgosh token ose mesazh thjeshtë return res.json({ success: true, token: "ok123" }); } else { console.log("❌ Gabim në kredenciale"); return res.status(401).json({ error: "Invalid credentials" }); } }); // Default route për të siguruar që index.html shfaqet app.get("/", (req, res) => { res.sendFile(path.join(__dirname, "index.html")); }); // 🚀 Start server (Render përdor PORT automatikisht) const PORT = process.env.PORT || 3000; app.listen(PORT, () => console.log(`✅ Server running on port ${PORT}`));
